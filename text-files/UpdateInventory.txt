<template>
    <div class="container mx-auto px-4 py-8">
        <h1 class="text-xl sm:text-2xl font-bold mb-6">Update Inventory</h1>

        <!-- Search and Filter -->
        <div class="mb-6 flex flex-wrap gap-4">
            <input v-model="searchTerm" placeholder="Search products..." class="p-2 border rounded" />
            <select v-model="categoryFilter" class="p-2 border rounded">
                <option value="">All Categories</option>
                <option v-for="category in flattenedCategories" :key="category._id" :value="category._id">
                    {{ category.name }}
                </option>
            </select>
            <button @click="applyFilters" class="bg-blue-500 text-white px-4 py-2 rounded hover:bg-blue-600">
                Apply Filters
            </button>
        </div>

        <!-- Inventory Table -->
        <div v-if="products.length > 0" class="overflow-x-auto">
            <table class="min-w-full bg-white">
                <thead class="bg-gray-100">
                    <tr>
                        <th class="px-4 py-2">Product</th>
                        <th class="px-4 py-2">Price</th>
                        <th class="px-4 py-2">Stock</th>
                        <th class="px-4 py-2">Category</th>
                        <th class="px-4 py-2">Discount</th>
                        <th class="px-4 py-2">Actions</th>
                    </tr>
                </thead>
                <tbody>
                    <tr v-for="product in filteredProducts" :key="product._id" class="border-b">
                        <td class="px-4 py-2">{{ product.name }}</td>
                        <td class="px-4 py-2">{{ formatPrice(product.price) }}</td>
                        <td class="px-4 py-2">
                            <input v-model.number="product.stock" type="number" min="0"
                                class="p-1 border rounded w-20" />
                        </td>
                        <td class="px-4 py-2">{{ getCategoryName(product.category) }}</td>
                        <td class="px-4 py-2">{{ product.discount }}%</td>
                        <td class="px-4 py-2">
                            <button @click="updateProduct(product)"
                                class="bg-green-500 text-white px-2 py-1 rounded hover:bg-green-600 mr-2">
                                Update
                            </button>
                            <button @click="editProduct(product)"
                                class="bg-blue-500 text-white px-2 py-1 rounded hover:bg-blue-600">
                                Edit
                            </button>
                        </td>
                    </tr>
                </tbody>
            </table>
        </div>
        <div v-else class="text-center py-4">
            Loading products...
        </div>
        <!-- Bulk Update Section -->
        <div class="mt-8">
            <h2 class="text-lg sm:text-xl font-semibold mb-4">Bulk Update</h2>
            <div class="flex flex-col sm:flex-row items-center gap-4">
                <select v-model="bulkUpdateAction" class="p-2 border rounded w-full sm:w-auto">
                    <option value="increase">Increase Stock</option>
                    <option value="decrease">Decrease Stock</option>
                    <option value="set">Set Stock</option>
                </select>
                <input v-model.number="bulkUpdateValue" type="number" min="0" class="p-2 border rounded w-full sm:w-24"
                    placeholder="Amount" />
                <button @click="applyBulkUpdate"
                    class="bg-purple-500 text-white px-4 py-2 rounded hover:bg-purple-600 w-full sm:w-auto">
                    Apply Bulk Update
                </button>
            </div>
        </div>

        <!-- CSV Import/Export -->
        <div class="mt-8 flex flex-col sm:flex-row gap-4">
            <button @click="exportCSV"
                class="bg-indigo-500 text-white px-4 py-2 rounded hover:bg-indigo-600 w-full sm:w-auto">
                Export CSV
            </button>
            <label
                class="bg-orange-500 text-white px-4 py-2 rounded hover:bg-orange-600 cursor-pointer text-center w-full sm:w-auto">
                Import CSV
                <input type="file" @change="importCSV" accept=".csv" class="hidden" />
            </label>
        </div>
    </div>
</template>

<script>
import { ref, computed, onMounted } from 'vue';
import { useProductStore } from '../store/productStore';
import { storeToRefs } from 'pinia';
import { toast } from 'vue-sonner';

export default {
    name: 'UpdateInventory',
    setup() {
        const productStore = useProductStore();
        const { products, flattenedCategories } = storeToRefs(productStore);

        // Initialize products as an empty array if it's undefined
        if (!products.value) {
            products.value = [];
        }

        const categories = ref([]);
        const searchTerm = ref('');
        const categoryFilter = ref('');
        const bulkUpdateAction = ref('increase');
        const bulkUpdateValue = ref(0);

        const fetchProducts = async () => {
            try {
                await productStore.fetchProducts();
                // Ensure products is an array after fetching
                if (!Array.isArray(products.value)) {
                    products.value = [];
                }
                products.value.forEach(product => product.newStock = product.stock);
            } catch (error) {
                toast.error('Failed to fetch products');
                products.value = []; // Set to empty array on error
            }
        };

        const fetchCategories = async () => {
            try {
                categories.value = await productStore.fetchCategories();
            } catch (error) {
                toast.error('Failed to fetch categories');
                categories.value = []; // Set to empty array on error
            }
        };

        const filteredProducts = computed(() => {
            if (!Array.isArray(products.value)) {
                return [];
            }
            return products.value.filter(product =>
                product.name.toLowerCase().includes(searchTerm.value.toLowerCase()) &&
                (categoryFilter.value === '' || product.category === categoryFilter.value)
            );
        });

        onMounted(async () => {
            try {
                await Promise.all([
                    fetchProducts(),
                    fetchCategories()
                ]);
            } catch (error) {
                toast.error('Failed to fetch data');
            }
        });

        const formatPrice = (price) => {
            return new Intl.NumberFormat('en-NG', { style: 'currency', currency: 'NGN' }).format(price);
        };

        const getCategoryName = (categoryId) => {
            const category = categories.value.find(cat => cat._id === categoryId);
            return category ? category.name : 'Uncategorized';
        };

        const applyFilters = () => {
            // The computed property will handle the filtering
        };

        const updateProduct = async (product) => {
            try {
                await productStore.updateProduct(product._id, { stock: product.stock });
                toast.success('Product updated successfully');
            } catch (error) {
                toast.error('Failed to update product');
            }
        };

        const editProduct = (product) => {
            // Implement edit functionality (e.g., open a modal)
            console.log('Editing product:', product);
        };

        const applyBulkUpdate = async () => {
            try {
                for (const product of filteredProducts.value) {
                    let newStock;
                    switch (bulkUpdateAction.value) {
                        case 'increase':
                            newStock = product.stock + bulkUpdateValue.value;
                            break;
                        case 'decrease':
                            newStock = Math.max(0, product.stock - bulkUpdateValue.value);
                            break;
                        case 'set':
                            newStock = bulkUpdateValue.value;
                            break;
                    }
                    await productStore.updateProduct(product._id, { stock: newStock });
                    product.stock = newStock;
                    product.newStock = newStock;
                }
                toast.success('Bulk update completed');
            } catch (error) {
                toast.error('Failed to apply bulk update');
            }
        };

        const exportCSV = () => {
            const csvContent = [
                ['Product ID', 'Name', 'Category', 'Stock'],
                ...filteredProducts.value.map(product => [
                    product._id,
                    product.name,
                    getCategoryName(product.category),
                    product.stock
                ])
            ].map(row => row.join(',')).join('\n');

            const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
            const link = document.createElement('a');
            if (link.download !== undefined) {
                const url = URL.createObjectURL(blob);
                link.setAttribute('href', url);
                link.setAttribute('download', 'inventory.csv');
                link.style.visibility = 'hidden';
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
            }
        };

        const importCSV = (event) => {
            const file = event.target.files[0];
            const reader = new FileReader();
            reader.onload = async (e) => {
                const text = e.target.result;
                const rows = text.split('\n').slice(1); // Skip header row
                for (const row of rows) {
                    const [id, , , stock] = row.split(',');
                    if (id && stock) {
                        try {
                            await productStore.updateProduct(id.trim(), { stock: parseInt(stock.trim(), 10) });
                        } catch (error) {
                            console.error(`Failed to update product ${id}:`, error);
                        }
                    }
                }
                await fetchProducts();
                toast.success('CSV import completed');
            };
            reader.readAsText(file);
        };

        return {
            products,
            categories,
            searchTerm,
            categoryFilter,
            bulkUpdateAction,
            bulkUpdateValue,
            filteredProducts,
            getCategoryName,
            applyFilters,
            applyBulkUpdate,
            exportCSV,
            importCSV,
            flattenedCategories,
            formatPrice,
            updateProduct,
            editProduct,
        };
    }
};
</script>
